\subsection*{(a)}
Show that $|r| = \sqrt{r\cdot r}$. Firstly, let $r = x\hat{x} + y \hat{y} + z \hat{z}$.
\begin{align*}
r =& x\hat{x} + y \hat{y} + z \hat{z}
\\
r\cdot r =& (x\hat{x} + y \hat{y} + z \hat{z})(x\hat{x} + y \hat{y} + z \hat{z})
\\
r\cdot r =& x^2 + y^2 + z^2 = r^2
\\
r\cdot r =& r^2
\\
\sqrt{r\cdot r} =& \sqrt{r^2}
\\
\sqrt{r\cdot r} =& |r|
\\
&\text{QED}
\end{align*}
